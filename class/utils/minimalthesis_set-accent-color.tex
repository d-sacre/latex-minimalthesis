%\@mt@visualStyling@colorScheme@setAccentColor
\newcommand{\mtVisualStylingColorSchemeSetAccentColor}[3]{%
    \ifthenelse{%
        \equal{\pgfkeysvalueof{/mt/visual styling/color scheme/type}}{plain}%
    }{%
        % plain color scheme
        % nothing to do
    }{%
        \ifthenelse{
            \equal{\pgfkeysvalueof{/mt/visual styling/color scheme/type}}{fancy-colorful}%
        }{%
            % fancy-colorful scheme
            \ifthenelse{
                \equal{\pgfkeysvalueof{/mt/visual styling/color scheme/colors/accent/#1/define/color model}}{}
            }{%
                % user did not specify color with model and value
                \ifthenelse{%
                    \equal{\pgfkeysvalueof{/mt/visual styling/color scheme/colors/accent/#1/set}}{}%
                }{%
                    % user did not specify any color
                }{%
                    % user did specify LaTeX internal color name or mix
                    \colorlet{#2}{\pgfkeysvalueof{/mt/visual styling/color scheme/colors/accent/#1/set}}
                }
            }{%
                % user specified color with model and value
                \definecolor{#2}{%
                    \pgfkeysvalueof{/mt/visual styling/color scheme/colors/accent/#1/define/color model}%
                }{%
                    \pgfkeysvalueof{/mt/visual styling/color scheme/colors/accent/#1/define/color value}%
                }%
            }%
        }{%
            % fancy-grayscale theme
            \colorlet{#2}{#3}%
        }%
    }%
} % #1 = pgfkeys only color, #2 = color name, #3 = grayscale color
